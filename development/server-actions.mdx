## Overview

Server actions are server-side functions that handle form submissions and data mutations. They're defined with the `"use server"` directive and are organized in `actions.ts` files within each feature module.

## File Organization

Server actions are colocated with the pages that use them:

```
app/(app)/
├── actions.ts              # Root-level actions (auth, branch switching)
├── branches/
│   └── actions.ts          # Branch CRUD operations
├── customers/
│   └── actions.ts          # Customer CRUD operations
├── inventory/
│   └── actions.ts          # Inventory management
├── vendors/
│   └── actions.ts          # Vendor management
└── purchases/
    └── actions.ts          # Purchase operations
```

## Basic Pattern

All server action files follow this structure:

```typescript
"use server";

import { revalidatePath } from "next/cache";
import { createClient } from "@/utils/supabase/server";

export async function createResource(formData: FormData) {
  const supabase = await createClient();
  
  // 1. Extract and validate form data
  const field = String(formData.get("field") ?? "").trim();
  
  if (!field) {
    throw new Error("Field is required");
  }
  
  // 2. Perform database operation
  const { error } = await supabase.from("table").insert({ field });
  
  if (error) {
    throw error;
  }
  
  // 3. Revalidate affected pages
  revalidatePath("/path");
  
  return { success: true } as const;
}
```

### Inventory Management with Context

From `app/(app)/inventory/actions.ts:177`, showing use of app context:

```typescript
"use server";

import { revalidatePath } from "next/cache";
import { redirect } from "next/navigation";
import { createClient } from "@/utils/supabase/server";
import { getAppContext } from "@/lib/server/app-context";

export async function createItem(formData: FormData) {
  const { activeBranch } = await getAppContext();
  if (!activeBranch) {
    throw new Error("Cannot create item without an active branch");
  }

  const supabase = await createClient();

  const name = String(formData.get("name") ?? "").trim();
  const code = String(formData.get("code") ?? "").trim();
  const categoryId = String(formData.get("categoryId") ?? "").trim() || null;
  const description = String(formData.get("description") ?? "").trim() || null;
  const sku = String(formData.get("sku") ?? "").trim();
  const sizeLabel = String(formData.get("size_label") ?? "").trim();
  const colorId = String(formData.get("color_id") ?? "").trim() || null;
  const baseCost = parseNumber(formData.get("base_cost"));
  const basePrice = parseNumber(formData.get("base_price"));

  if (!name || !code || !sku || !sizeLabel) {
    throw new Error("Name, code, sku, and size are required");
  }

  // Create item
  const { data: itemData, error: itemError } = await supabase
    .from("item")
    .insert({
      name,
      code,
      category_id: categoryId,
      description,
    })
    .select("id")
    .single();

  if (itemError) {
    throw itemError;
  }

  // Create variant
  const { error: variantError } = await supabase.from("item_variant").insert({
    item_id: itemData.id,
    sku,
    size_label: sizeLabel,
    color_id: colorId,
    base_cost: baseCost,
    base_price: basePrice,
  });

  if (variantError) {
    throw variantError;
  }

  revalidatePath("/inventory");
  redirect("/inventory");
}
```

### Authentication Actions

From `app/(auth)/login/actions.ts`:

```typescript
"use server";

import { redirect } from "next/navigation";
import { createClient } from "@/utils/supabase/server";

type SignInState = {
  error?: string;
};

const EMAIL_REGEX = /.+@.+\..+/;

function parseCredentials(formData: FormData) {
  const identifier = String(formData.get("identifier") ?? "").trim();
  const password = String(formData.get("password") ?? "").trim();

  if (!identifier) {
    return { error: "Please enter your email or mobile number." } as const;
  }

  if (!password) {
    return { error: "Please enter your password." } as const;
  }

  if (password.length < 6) {
    return { error: "Password must be at least 6 characters long." } as const;
  }

  if (EMAIL_REGEX.test(identifier)) {
    return { email: identifier, password } as const;
  }

  const phone = identifier.replace(/\s+/g, "");
  if (!/^[0-9+]+$/.test(phone)) {
    return {
      error: "Enter a valid phone number (digits and optional + only).",
    } as const;
  }

  return { phone, password } as const;
}

export async function signIn(_: SignInState, formData: FormData): Promise<SignInState> {
  const credentials = parseCredentials(formData);

  if ("error" in credentials) {
    return { error: credentials.error };
  }

  const supabase = await createClient();

  const { error } = await supabase.auth.signInWithPassword(credentials);

  if (error) {
    return { error: error.message };
  }

  redirect("/dashboard");
}
```

function parseNumber(value: FormDataEntryValue | null) {
  if (typeof value !== "string") return null;
  const parsed = Number.parseFloat(value);
  return Number.isFinite(parsed) ? parsed : null;

function normaliseHex(value: string) {
  const trimmed = value.trim();
  if (!trimmed) return null;
  if (/^#[0-9a-fA-F]{6}$/.test(trimmed)) {
    return trimmed.toUpperCase();
  }
  const withHash = `#${trimmed.replace(/^#/, "")}`;
  return /^#[0-9a-fA-F]{6}$/.test(withHash) ? withHash.toUpperCase() : null;
}

const hexValue = normaliseHex(String(formData.get("hex_value") ?? ""));

## Best Practices

1. **Always use "use server"** - Place at the top of action files
2. **Validate input** - Check required fields before database operations
3. **Handle errors** - Throw or return errors for client handling
4. **Revalidate paths** - Update cache after mutations
5. **Use FormData** - Accept `FormData` parameter for form integration
6. **Type return values** - Use `as const` for success objects
7. **Colocate actions** - Keep actions with related pages
8. **Trim strings** - Always trim string inputs
9. **Null vs undefined** - Use `null` for optional database fields
10. **Use app context** - Load branch/user context when needed

## Security

- Server actions only run on the server (never exposed to client)
- Authentication is enforced through `getAppContext()` in protected actions
- Validation prevents invalid data from reaching the database
- Supabase Row Level Security provides additional authorization layer
